    Openvidu 视频会议

    jfwang 2022/5/29

    This is Openvidu.

    # deploy in docker

    # 您需要 root 权限才能部署 OpenVidu。
su

# 安装 OpenVidu 的推荐文件夹是 /opt。文档中有关本地部署的所有其他说明都采用此安装路径。
cd /opt

# 现在执行以下命令下载并运行安装脚本。
curl https://s3-eu-west-1.amazonaws.com/aws.openvidu.io/install_openvidu_latest.sh | bash

如果下载失败：openvidu Error when downloading the file '.env' 这是github连接失败，需要配置hosts
找到多地区ping网站，找到可用的 raw.githubusercontent.com 对应的IP
vim /etc/hosts

185.199.108.133   raw.githubusercontent.com
185.199.109.133   raw.githubusercontent.com

# OpenVidu 平台配置在带有环境变量的 .env 文件中指定
OpenVidu 平台配置在带有环境变量的 .env 文件中指定。

必须为属性 DOMAIN_OR_PUBLIC_IP 和 OPENVIDU_SECRET 指定值。默认空值将失败。
如果您拥有有效的域名，则可以更改CERTIFICATE_TYPE。将此属性设置为将自动为您生成有效的证书（需要设置属性）。
或者，如果出于任何未知原因，您更喜欢使用自己的证书，请将属性设置为并放置证书文件，并按说明进行放置。letsencryptLETSENCRYPT_EMAILowncert

# OpenVidu 平台配置在带有环境变量的 .env 文件中指定
cd /opt/openvidu
vim .env

DOMAIN_OR_PUBLIC_IP = 120.48.97.128
OPENVIDU_SECRET = 123456

# 运行
./openvidu start

# 访问 https://120.48.97.128/
user:     admin
password: 123456

    # 最小量部署openvidu

    # 假设openvidu MY_SECRET = 123456

# 查看本机IP
ip -4 -oneline route get 1.0.0.0 | grep -Po 'src \K\S+'

# 创建openvidu-server容器
docker run -p 4443:4443 --name openvidu-server \
  -e OPENVIDU_SECRET=MY_SECRET \
  -e DOMAIN_OR_PUBLIC_IP=$(ip -4 -oneline route get 1.0.0.0 | grep -Po 'src \K\S+') \
  -e SERVER_PORT=4443 \
  -e HTTPS_PORT=443 \
  -e FORCE_PLAIN_HTTP=false \
  -d \
openvidu/openvidu-dev:2.29.0

# 可通过 OpenVidu Call[10] 应用来验证 OpenVidu 服务部署结果。
docker run -p 5000:5000 --name openvidu-call  \
  -e OPENVIDU_URL=https://$(ip -4 -oneline route get 1.0.0.0 | grep -Po 'src \K\S+') \
  -e OPENVIDU_SECRET=123456 \
  -d \
openvidu/openvidu-call:2.29.0

# 此时，可以通过 http://IP:5000 访问到 OpenVidu Call 应用界面。WebRTC 相关的语音、视频等功能需部署 Proxy 组件后通过 HTTPS 协议访问才能正常使用。
http://IP:5000
http://172.30.124.69:5000


# Proxy 组件的作用，是允许通过 HTTPS 访问应用。本地环境可以借助 openvidu-tutorials/certs[12] 中提供的证书，
# 以及将 openvidu-tutorials/openvidu-angular/nginx.conf 第 12 行[13] 端口由 4200 修改为 5000
docker run --name nginx-openvidu -p 443:443 \
  --add-host=host.docker.internal:$(ip -4 -oneline route get 1.0.0.0 | grep -Po 'src \K\S+') \
  -v /data/nginx-openvidu/nginx.conf:/etc/nginx/nginx.conf:ro \
  -v /data/nginx-openvidu/certs:/etc/nginx/certs:ro \
  -d \
nginx
