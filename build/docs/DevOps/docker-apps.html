    # 安装rabbitmq

    docker pull rabbitmq
docker run -d --name rabbitmq -p 15672:15672 -p 5672:5672 rabbitmq
docker exec -it rabbitmq bash
rabbitmq-plugins enable rabbitmq_management # 开启网页管理界面

#RabbitMQ 后台管理页面上开启 Channels
cd  /etc/rabbitmq/conf.d/
echo management_agent.disable_metrics_collector = false > management_agent.disable_metrics_collector.conf
docker restart rabbitmq

    # 安装Nexus

    1. 拉取镜像
   docker pull sonatype/nexus3:latest
2. 创建挂载目录
   mkdir /data/nexus-data
   chown -R 200 /data/nexus-data
3. 运行
   docker run -d -p 8081:8081 --name nexus \
   -e INSTALL4J_ADD_VM_PARAMS="-Xms512m -Xmx512m \
   -XX:MaxDirectMemorySize=1200m" \
   -v /data/nexus-data:/nexus-data sonatype/nexus3
# -f 实时追加最新日志
# --tail=N 查看日志的最后 N 行
docker logs -f --tail=100 nexus
进入主页面：
http://localhost:8081/

三、用户登录
nexus3的账号是：admin
密码要去挂载的目录查看一个叫admin.password的文件，里面放的就是admin的登录密码。
cd /data/nexus-data
查看密码
cat admin.password

    # docker安装elasticsearch

    # 参考 https://www.cnblogs.com/luliang888/p/17583235.html
拉取镜像  --7.17.3
docker pull elasticsearch:7.17.3

第一次执行docker脚本
docker run -it \
    -p 9200:9200 \
    -p 9300:9300 \
    --name es \
    --net es-net \
    -e "ES_JAVA_OPTS=-Xms512m -Xmx512m" \
    -e "discovery.type=single-node" \
    -e LANG=C.UTF-8 \
    -e LC_ALL=C.UTF-8 \
    elasticsearch:7.17.3
    
   
创建Elasticsearch挂载目录 
mkdir -p /data/elasticsearch

将容器内的文件复制到主机上
docker cp es:/usr/share/elasticsearch/config /data/elasticsearch
docker cp es:/usr/share/elasticsearch/data /data/elasticsearch
docker cp es:/usr/share/elasticsearch/plugins /data/elasticsearch
docker cp es:/usr/share/elasticsearch/logs /data/elasticsearch


删除容器
docker rm -f es

修改docker脚本，增加-v挂载目录和-d参数
docker run -d \
    --name es \
    --privileged \
    --network es-net \
    -e "ES_JAVA_OPTS=-Xms512m -Xmx512m" \
    -e "discovery.type=single-node" \
    -e LANG=C.UTF-8 \
    -e LC_ALL=C.UTF-8 \
    -p 9200:9200 \
    -p 9300:9300 \
    -v es-data:/data/elasticsearch/data \
    -v es-plugins:/data/elasticsearch/plugins \
    -v es-logs:/data/elasticsearch/logs \
elasticsearch:7.17.3

设置修改账号密码
docker exec -it es /bin/bash
cd /usr/share/elasticsearch/bin
./elasticsearch-setup-passwords interactive

登录
https://127.0.0.1:9200/

    # docker安装nginx

    # 创建挂载
mkdir -p /data/nginx/conf
mkdir -p /data/nginx/app
mkdir -p /data/nginx/logs

# 拉取镜像
docker pull nginx

# 创建模板nginx容器
docker run -d -p 8500:8500 --name nginx nginx

# 拷贝nginx.conf
docker cp nginx:/etc/nginx/nginx.conf /workspace/data/nginx-iboot/nginx.conf

# 创建nginx容器
docker run -d -p 8500:8500 --name nginx-iboot \
-v /data/nginx/nginx.conf:/etc/nginx/nginx.conf \
-v /data/nginx/conf:/etc/nginx/conf.d \
-v /data/nginx/app:/usr/share/nginx/html \
-v /data/nginx/logs:/var/log/nginx \
nginx


docker run -d -p 81:81 --name nginx-iblog \
-v /data/nginx-iblog/nginx.conf:/etc/nginx/nginx.conf \
-v /data/nginx-iblog/conf:/etc/nginx/conf.d \
-v /data/nginx-iblog/app:/usr/share/nginx/html \
-v /data/nginx-iblog/logs:/var/log/nginx \
nginx

docker run -d -p 82:82 --name nginx-media \
-v /data/nginx-media/nginx.conf:/etc/nginx/nginx.conf \
-v /data/nginx-media/media:/usr/share/nginx/html \
-v /data/nginx-media/logs:/var/log/nginx \

    # docker安装mysql

    # 创建挂载目录
mkdir /workspace/data/mysql8/mysql
mkdir /workspace/data/mysql8/logs
mkdir /workspace/data/mysql8/data

# 创建临时容器
docker run -p 3336:3306 --name mysql86 -e MYSQL_ROOT_PASSWORD=haihu123456 -d mysql

# 拷贝临时容器内容到
docker cp mysql86:/etc/mysql /workspace/data/mysql8/

# 创建容器
docker run \
-p 3316:3306 \
--name mysql8 \
--privileged=true \
-v /workspace/data/mysql8/mysql:/etc/mysql \
-v /workspace/data/mysql8/logs:/logs \
-v /workspace/data/mysql8/data:/var/lib/mysql \
-v /etc/localtime:/etc/localtime \
-e MYSQL_ROOT_PASSWORD=haihu123456 \
-d mysql

# 授权远程访问
grant all privileges on *.* to root@'%' with grant option;
# 刷新权限
flush privileges;

    # 安装nacos

    # 如果没有则创建网络
docker network create pub_net

# 将mysql8容器 加入到 pub_net 网络中
docker network connect pub_net mysql8

# 查看mysql8的信息（包括网络信息）
docker inspect mysql8

# 创建文件挂载
mkdir -p /data/nacos/logs
mkdir -p /data/nacos/conf

# 获取镜像
docker pull nacos/nacos-server

# 创建nacos临时容器
docker run -p 8884:8848 --name nacos2 -d nacos/nacos-server

# 从临时容器拷贝内容
docker cp nacos2:/home/nacos/logs /data/nacos
docker cp nacos2:/home/nacos/conf /data/nacos
#删除临时容器
docker stop nacos2
docker rm nacos2

# 修改配置文件
cd /data/nacos/conf
vim application.properties
在【db.url.0】 的值后拼接 `&allowPublicKeyRetrieval=true`

# 创建正式容器
docker run --env MODE=standalone --name nacos \
--network=pub_net \
-v /data/nacos/logs:/home/nacos/logs \
-v /data/nacos/conf:/home/nacos/conf \
-e SPRING_DATASOURCE_PLATFORM=mysql \
-e MYSQL_DATABASE_NUM=1 \
-e MYSQL_SERVICE_HOST=mysql8 \
-e MYSQL_SERVICE_PORT=3306 \
-e MYSQL_SERVICE_DB_NAME=nacos_config \
-e MYSQL_SERVICE_USER=root \
-e MYSQL_SERVICE_PASSWORD=wangjiafu \
-e JVM_XMS=256m \
-e JVM_XMX=256m \
-d -p 8848:8848 nacos/nacos-server

# 访问
http://localhost:8848/nacos
user: nacos
password: nacos
