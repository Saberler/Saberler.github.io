    # 查看网络信息

    ifconfig 或 ip addr show
每个网络称为一个网络接口（interface）

物理网卡一般以 eth、ens开头
lo为环回接口 loop
v 开头一般为虚拟网卡

    # 网卡与网口的对应

    如果服务器有多个网线接口，需要确认哪个网口再系统里对应的网卡名称
使用ethtool命令确认

cd /etc/sysconfig/network-scripts/
找到 ifcfg- 开头的，后面的字符就是网卡名称

执行 ethtool -p [inet-name] ，命令会一直处于运行状态
到服务器物理接口处查看指示灯，按某一频率一直闪烁的口，且按ctrl+c后停止频率闪烁的网口就是目标网口。

    # 修改网络配置

    cd /etc/sysconfig/network-scripts/
vim ifcfg-[inet-name]编辑目标网卡

配置如下：

NAME=ens33 #不能动!!!
UUID=7ea6b77b-6f70-4768-9990-7e7dcc6140b1 #不能动!!!
DEVICE=ens33 #不能动!!!
TYPE=Ethernet
PROXY_METHOD=none
BROWSER_ONLY=no
BOOTPROTO=static   #dhcp or static
DEFROUTE=yes
IPV4_FAILURE_FATAL=no
IPV6INIT=yes
IPV6_AUTOCONF=yes
IPV6_DEFROUTE=yes
IPV6_FAILURE_FATAL=no
IPV6_ADDR_GEN_MODE=stable-privacy
ONBOOT=yes  	#--这里不需要修改, 开机自启动
IPADDR=192.168.200.101	#--这里改成所需要的网址
PREFIX=24	#--这里改成24位，相当于掩码是255.255.255.0
#GATEWAY=192.168.200.2  #--网关，配置可以联网
#DNS1=192.168.0.1
#DNS2=192.168.0.2

配置修改完成之后重启网络：
systemctl restart network

    # 给内网/离线服务器配置互联网

    windows电脑通过网线与服务器网口连接，需要确认服务器网口对应的服务器网卡名称
windows打开网络适配面板，将可以访问互联网的网络分享给服务器，一般是名为wifi的网络：属性-》共享-》以太网（通过网线与服务器直连，一般就是以太网）
服务器需要设置网卡为自动获取IP
vim /etc/sysconfig/network-scripts/ifcfg-eth1
改成：
NAME=eth0 #不能动!!!
UUID=7ea6b77b-6f70-4768-9990-7e7dcc6140b1 #不能动!!!
DEVICE=eth0 #不能动!!!
BOOTPROTO=dhcp
ONBOOT=yes
STARTMODE=auto
TYPE=Ethernet
USERCTL=no

重启网络：
systemctl restart network

测试:
ping 公网IP
ping 8.8.8.8

    # 防火墙

    # 查看防火墙状态
systemctl status firewalld
service firewalld status

# 开启防火墙
systemctl start firewalld

# 关闭防火墙
systemctl stop firewalld

# 防火墙开机启动
systemctl enable firewalld

# 防火墙开放端口
防火墙开放端口centos
在CentOS系统中，可以使用firewall-cmd命令来配置防火墙，以开放特定的端口。以下是开放端口的步骤和示例代码：

# 查看当前活动的防火墙区域配置：
firewall-cmd --state

#列出所有打开的端口：
firewall-cmd --list-ports

# 开放指定的端口（以开放端口8080为例）：
firewall-cmd --zone=public --add-port=8080/tcp --permanent

# 重新加载防火墙以应用更改：
firewall-cmd --reload

# 确认端口已成功开放：
firewall-cmd --list-ports

# 请注意，如果你的CentOS系统使用的是iptables而不是firewalld，那么你需要使用iptables命令来管理端口规则。

    # 配置root ssh登录

    # 查看ssh服务是否开启
systemctl status sshd

# 编辑配置文件
vim /etc/ssh/sshd_config

# 设置允许root登录
PermitRootLogin yes

# 设置允许使用密码登录
PasswordAuthentication yes

# 配置此项之后，就只能是root或guest用户ssh登录，其他用户不允许
# 去掉这个配置，则所有用户都可以登录
AllowUsers root guest

# 重启sshd服务
systemctl restart sshd

    # centos 服务器密码忘记-单机维护模式

    centos忘记root密码
如果您忘记了CentOS系统的root密码，可以按照以下步骤来重置密码：

重启系统，在启动时进入GRUB菜单。

在启动选项列表中选择要启动的内核版本，然后按e键编辑启动参数。

找到以linux16开头的行，更改ro（只读）为 rw init=/sysroot/bin/sh

按 Ctrl + X 启动系统。

系统启动后，会进入单用户模式。执行以下命令来重新挂载根文件系统：

chroot /sysroot
LANG=en                 #更改语言为en，改密码时才不会显示乱码
passwd root
touch /.autorelabel
更改密码后，执行 exit 退出chroot环境，然后重启系统：
reboot -f
请注意，在执行这些步骤时，您需要具备物理访问权限或远程KVM等访问权限，并且在执行重启命令时使用-f参数可以强制重启系统。
